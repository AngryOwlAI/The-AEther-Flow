\begin{abstract}
This draft/control artifact specifies a source-forcing discharge protocol and
controlled-pause ledger for the selector-domain source-basis path. It does
not discharge the six theorem burdens. It classifies what a future discharge
must do in the current registered source language, what negative-control
families must be retained, what proof attempts require protected authority
expansion, and which statuses must remain \(\mathsf{not\_computed}\) or
\(\mathsf{underdetermined}\). The six surfaces are ledger provenance,
horizon universality, admissibility meta-grammar, certificate invariance,
output-basis generation, and \(D_F\) status transitions. Candidate
reconstruction, canonical ontology edit, benchmark promotion, Gate Chair
review, and effective-sector recovery claims remain blocked.
\end{abstract}

\section{Scope and Protocol Standard}

This artifact is a draft/control output for task \texttt{RT-20260613-006}.
It responds to the source-forcing obligation packet and its smuggling audit.
The audit found no direct target primitive in the obligation packet, but it
also found that the obligations are theorem burdens rather than theorem
discharges. This protocol therefore defines a ledger for future work rather
than a promotion step.

Let \(\Lang\) denote the registered source language before the A0--A4
extension sequence. For each choice surface
\[
X\in\{H,\HorizonFam,\EnumFam_i,\CertFam_i,\OutFam_i,D_F\},
\]
the discharge record has four cells:
\[
\mathcal{D}_X=(\Current_X,\Neg_X,\Expand_X,\Unres_X).
\]
Here \(\Current_X\) is the current-language theorem burden, \(\Neg_X\) is the
negative-control family that must be retained, \(\Expand_X\) is the protected
authority-expansion trigger, and \(\Unres_X\) is the status that must remain
when the burden is not discharged.

\begin{definitionblock}[Discharge eligibility]
A surface \(X\) is eligible for later candidate-support consideration only
when a source-side rule \(\Force_X:\Lang\to X\) is accompanied by a theorem
showing that every \(\Ad\)-admissible and \(\Blind\)-blind alternative is
source-equivalent, source-impossible, or retained as a negative control whose
outputs and statuses are not projected into candidate support.
\end{definitionblock}

\begin{protocolrule}[No inferential upgrade]
This protocol may not upgrade any surface to \(\mathsf{invariant}\). In the
absence of a registered current-language theorem, a surface may move only to
\(\mathsf{family\_valued}\), \(\mathsf{underdetermined}\), or
\(\mathsf{not\_computed}\). If a proof needs canonical ontology authority,
benchmark-status authority, Gate Chair authority, or permanent role
registration, the correct record is \(\Expand_X\), not a local discharge.
\end{protocolrule}

\section{Ledger Provenance}

\begin{ledgerentry}[Current-language theorem burden]
\(\Current_H\) must prove that the provenance tuple
\[
\Pi_H=(\mathcal{R}_{0},\mathcal{C}_{0},\mathcal{O}_{0},
\mathcal{Q}_{0},\mathcal{B}_{0})
\]
is generated from registered source records, source closure rules, source
operation seeds, source quotient seeds, and pre-benchmark blinding records
already expressible in \(\Lang\). The proof must establish uniqueness up to
source relabeling or complete enumeration of all admissible ledgers in
\(\LedgerFam\).
\end{ledgerentry}

\begin{ledgerentry}[Negative-control family]
\(\Neg_H\) consists of all admissible ledgers generated from the same source
records under broader, narrower, or incomparable source closure and quotient
rules. If these ledgers change presentation boundaries, fronts, ranks,
spectra, transport reports, or status rows, the full family remains recorded.
\end{ledgerentry}

\begin{ledgerentry}[Protected expansion trigger]
\(\Expand_H\) is triggered if any component of \(\Pi_H\) requires a new
canonical source rule, a new ontology primitive, a benchmark-selected seed,
or target evidence such as target metric behavior, target locality, target
observer protocol, target proper-time normalization, target-favorable rank,
or benchmark success.
\end{ledgerentry}

\begin{ledgerentry}[Residual status]
If \(\Current_H\) is absent and \(\Neg_H\) is incomplete, dependent rows must
remain \(\mathsf{underdetermined}\) or \(\mathsf{not\_computed}\).
\end{ledgerentry}

\section{Horizon Universality}

\begin{ledgerentry}[Current-language theorem burden]
\(\Current_{\HorizonFam}\) must construct a source-generated horizon family
\[
\HorizonFam=\{B_\eta:\eta\in E\}
\]
from source counts, source-order depth, or source saturation rules, then
prove that every admissible horizon is included, source-impossible, or
tracked as unresolved. Any saturation predicate must be monotone under
admissible source extension.
\end{ledgerentry}

\begin{ledgerentry}[Negative-control family]
\(\Neg_{\HorizonFam}\) includes smaller, larger, and incomparable horizons
that remain source-admissible under the same source seeds. If extension of
the horizon creates adverse boundaries, fronts, spectra, cycles, or status
rows, those outputs must remain family-valued.
\end{ledgerentry}

\begin{ledgerentry}[Protected expansion trigger]
\(\Expand_{\HorizonFam}\) is triggered by any horizon cutoff justified by
target scale, target locality, target clock normalization, or benchmark
success. It is also triggered when a required saturation rule is not
expressible in the registered source language.
\end{ledgerentry}

\begin{ledgerentry}[Residual status]
Without universality and monotone saturation evidence, the horizon surface is
\(\mathsf{not\_computed}\) for candidate support and may be
\(\mathsf{family\_valued}\) for negative-control reporting.
\end{ledgerentry}

\section{Admissibility Meta-Grammar}

\begin{ledgerentry}[Current-language theorem burden]
For each \(D_i\), \(\Current_{\EnumFam_i}\) must define a source-side
meta-grammar
\[
\MetaG_i=(\Sigma_i,\mathcal{R}_i,\mathcal{E}_i,\mathcal{Q}_i,\mathcal{A}_i)
\]
that generates the relevant enumerator family from source symbols, rewrite or
closure rules, expansion rules, duplicate quotients, and adverse-output
retention rules. The proof must show completeness over narrower, broader,
and incomparable source-admissible grammars.
\end{ledgerentry}

\begin{ledgerentry}[Negative-control family]
\(\Neg_{\EnumFam_i}\) is the ledger of generated grammars whose outputs
disagree or whose output-basis status is non-singleton. These grammars may be
classified as duplicate, out of horizon, source invalid, authority expansion,
or not computed, but they may not be removed for candidate convenience.
\end{ledgerentry}

\begin{ledgerentry}[Protected expansion trigger]
\(\Expand_{\EnumFam_i}\) is triggered when grammar validity depends on a new
canonical source syntax, a new admissibility authority, a benchmark-selected
grammar family, or a relevance claim that is not itself a source theorem.
\end{ledgerentry}

\begin{ledgerentry}[Residual status]
If meta-grammar completeness is not proved, enumerator-dependent results
remain \(\mathsf{family\_valued}\), \(\mathsf{underdetermined}\), or
\(\mathsf{not\_computed}\).
\end{ledgerentry}

\section{Certificate Invariance}

\begin{ledgerentry}[Current-language theorem burden]
\(\Current_{\CertFam_i}\) must prove, for every admissible certificate pair
\(C,C'\in\CertFam_i\), that kept and omitted cases agree up to declared
source relabeling or that every disagreement is retained as a certificate-
sensitive negative control:
\[
C(x)\neq C'(x)\Rightarrow \Neg_i(x,C,C').
\]
\end{ledgerentry}

\begin{ledgerentry}[Negative-control family]
\(\Neg_{\CertFam_i}\) includes all cases where duplicate, out-of-horizon,
source-invalid, authority-expansion, or not-computed labels differ across
admissible certificate rules and affect any allowed output basis.
\end{ledgerentry}

\begin{ledgerentry}[Protected expansion trigger]
\(\Expand_{\CertFam_i}\) is triggered when an exclusion label requires a new
canonical certificate rule, target relevance, candidate convenience, or
unregistered authority to decide that an omitted case has no source effect.
\end{ledgerentry}

\begin{ledgerentry}[Residual status]
Replayable certificates without invariance theorem evidence do not support
\(\mathsf{invariant}\). The status remains \(\mathsf{underdetermined}\) or
\(\mathsf{family\_valued}\).
\end{ledgerentry}

\section{Output-Basis Generation}

\begin{ledgerentry}[Current-language theorem burden]
\(\Current_{\OutFam_i}\) must define a source-side generator
\[
\mathsf{Gen}_{O,i}:\Lang\times\EnumFam_i\times\CertFam_i\to\OutFam_i
\]
that records favorable, adverse, and neutral source quantities at the same
grain. The proof must show that presentation boundaries, support fronts,
discriminator ranks, quotient spectra, transport cycles, holonomy reports,
and status rows are source-definable or explicitly unresolved.
\end{ledgerentry}

\begin{ledgerentry}[Negative-control family]
\(\Neg_{\OutFam_i}\) contains broader output bases that expose non-singleton
or adverse source behavior. A favorable projection remains recorded as a
projection and cannot replace the broader source-admissible output family.
\end{ledgerentry}

\begin{ledgerentry}[Protected expansion trigger]
\(\Expand_{\OutFam_i}\) is triggered when rank, front, spectrum, transport,
holonomy, or status vocabulary is imported from target effective geometry or
requires a canonical ontology extension not already authorized.
\end{ledgerentry}

\begin{ledgerentry}[Residual status]
Without generator completeness and adverse-output retention, output-basis
results remain \(\mathsf{family\_valued}\) or \(\mathsf{underdetermined}\).
\end{ledgerentry}

\section{\texorpdfstring{\(D_F\)}{DF} Status Transitions}

\begin{ledgerentry}[Current-language theorem burden]
\(\Current_{D_F}\) must state a transition theorem for every move into
\(\mathsf{invariant}\). The theorem must cite ledger provenance, horizon
universality, meta-grammar completeness, certificate invariance,
output-basis generator completeness, and an independent target-import audit.
\end{ledgerentry}

\begin{ledgerentry}[Negative-control family]
\(\Neg_{D_F}\) includes status rows that change under admissible ledger,
horizon, grammar, certificate, output-basis, or evidence-threshold variation.
These rows remain explicit and may not be suppressed by selecting a favorable
status rule.
\end{ledgerentry}

\begin{ledgerentry}[Protected expansion trigger]
\(\Expand_{D_F}\) is triggered if invariant status depends on benchmark
success, target agreement, Gate Chair authority, canonical ontology edit, or
a permanent status-rule registration not authorized by this task.
\end{ledgerentry}

\begin{ledgerentry}[Residual status]
When any prerequisite theorem is absent, \(D_F\) may preserve
\(\mathsf{not\_computed}\), \(\mathsf{underdetermined}\), or
\(\mathsf{family\_valued}\). It may not move to \(\mathsf{invariant}\).
\end{ledgerentry}

\begin{pauserule}[Controlled pause rather than reconstruction]
If any row remains not discharged, the Director should preserve a controlled
pause for candidate reconstruction. This pause is local: it does not reject
the \TheoryProgram{} and does not forbid future source-side theorem work.
It prevents the current packet from being used as candidate support before
the required source-side evidence exists.
\end{pauserule}

    expressible in the current registered source language.
    \item Enumerate the relevant negative-control family before checking
    favorable projections.
    \item Attempt the current-language theorem for exactly one surface.
    \item If the theorem fails, preserve the negative-control family and
    record \(\mathsf{underdetermined}\), \(\mathsf{family\_valued}\), or
    \(\mathsf{not\_computed}\).
    \item If the theorem requires protected authority, stop and route the
    expansion request through the appropriate human-gated process.
    \item Only after a theorem is written may a Smuggling Auditor test the
    discharge chain for hidden target import and authority expansion.
\end{enumerate}
